- 코드 :
```c++
#include <iostream>

using namespace std;

int d[51][51];

int pascal(int a, int b);

int main()
{
    int a,b;
    cin>>a>>b;
    cout<<pascal(a-1,b-1)<<endl;
    return 0;
}

int pascal(int a, int b){
    if ((a==0)||(b==0)){
        return d[a][b]=1;
    }
    else if (d[a][b]!=0 && d[a][b]!=1) return d[a][b]%100000000;
    else{
        return d[a][b] = ( pascal(a-1,b)%100000000 + pascal(a,b-1)%100000000 )%100000000;
    }
}
```
